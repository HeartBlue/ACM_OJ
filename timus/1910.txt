#define MAX(a,b) ((a) > (b) ? (a) : (b))
#define MIN(a,b) ((a) < (b) ? (a) : (b))
#define MEM(a,b) memset((a),(b),sizeof(a))
const LL INF = 1000000007;
int a[10001];
int main()
{
	//freopen("input.txt", "r", stdin);
	int n;
	cin >> n;
	MEM(a,0);
	for(int i = 1; i <= n; i++) cin >> a[i];
	int maxv = 0;
	int key = -1;
	for(int i = 1; i <= n; i++)
	{
		if(a[i]+a[i-1]+a[i+1] > maxv)
		{
			maxv = a[i-1]+a[i]+a[i+1];
			key = i;
		}
	}
	cout << maxv << ' ' << key << endl;
	return 0;
}
