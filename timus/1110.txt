int powmod(int x, int n, int mod)
{
	int ret = 1%mod;
	for(int i = 0; i < n; i++)
	{
		ret *= x;
		ret %= mod;
	}
	return ret;
}

int main()
{
	//freopen("input.txt", "r", stdin);
	int n,m,y;
	cin >> n >> m >> y;
	vector<int> ans;
	for(int i = 0; i < m; i++)
	{
		if(powmod(i,n,m) == y)
		{
			ans.push_back(i);
		}
	}
	if(ans.empty()) cout << "-1" << endl;
	else
	{
		for(int i = 0; i < ans.size(); i++)
		{
			if(i) cout << ' ';
			cout << ans[i];
		}
		cout << endl;
	}

	return 0;
}
